        //2. 在函数中使用
        function sum(...numbers){
            console.log(numbers)//numbers是一个数组
            return numbers.reduce((total,current)=>{
                console.log("#",total);
                console.log(current);
                return total+current;
            })
        }
        console.log(sum(1,2,3))

        //3. 构造字面量对象时使用（复制对象）
        let person ={name:'tom',age:18};
        let person2 = person
        let person3 ={...person}
        //console.log(...person);//报错 展开运算符不能展开对象
        person.name="jerry";
        console.log(person2.name);//jerry
        console.log(person3.name);//tom
        
        //4. 合并
        let person4={...person,name:'jack'};
        console.log(person4);
